function ConcatObjects () {
	var resultObj = {};
	var i = 0;
	
	for  ( i = 0; i < arguments.length; i++ ) {
		if ( typeof arguments[i] == "object" ) {
			for (var attrname in arguments[i]) { 
				resultObj[attrname] = arguments[i][attrname]; 
			}	
		} 
	}
	return resultObj;
	
}
